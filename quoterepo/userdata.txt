#!/bin/bash
sudo su
curl -o- https://raw.githubusercontent.com/nvm-sh/nvm/v0.34.0/install.sh | bash
. ~/.nvm/nvm.sh
nvm install node
aws s3 cp "s3://asmit-work-bucket/deployment.tar.gz" deployment.tar.gz
gunzip deployment.tar.gz
tar -xf deployment.tar
cd serverSide
npm install -g pm2
pm2 start server.js
